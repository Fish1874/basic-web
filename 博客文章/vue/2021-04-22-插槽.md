### 默认插槽

```html
<!DOCTYPE html>
<html lang="en">

<head>
  <meta charset="UTF-8">
  <meta http-equiv="X-UA-Compatible" content="IE=edge">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Document</title>
  <link rel="stylesheet" href="https://unpkg.com/element-ui/lib/theme-chalk/index.css">
  <style>

  </style>
</head>

<body>
  <div id="app">
    <!------ 默认插槽 ------>
    <navigation-link1>我更改了子组件的默认内容</navigation-link1>
  </div>
</body>
<script src="https://cdn.bootcdn.net/ajax/libs/vue/2.6.9/vue.js"></script>
<script src="https://unpkg.com/element-ui/lib/index.js"></script>
<script>
  /**
   * 子组件
  */
  Vue.component('navigation-link', {
    data() {
      return {
      }
    },
    template: `
      <div>
        <h2>默认插槽</h2>
        <ul>
          <li>
            <p>【 默认显示子组件的内容,除非父组件更改了插槽 】</p>
            <slot>
              <el-radio>el-radio组件</el-radio>
              默认内容：学乃身之宝，儒为席上珍。
            </slot>
          </li>
        </ul>
      </div>
    `
  });

  var app = new Vue({
    el: '#app',
  });

</script>

</html>
```



### 具名插槽

```html
<!DOCTYPE html>
<html lang="en">

<head>
  <meta charset="UTF-8">
  <meta http-equiv="X-UA-Compatible" content="IE=edge">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Document</title>
  <link rel="stylesheet" href="https://unpkg.com/element-ui/lib/theme-chalk/index.css">
</head>

<body>
  <div id="app">

    <!------ 具名插槽 ------>
    <navigation-link>
      <template v-slot:header>
        <h2>基本使用：v-slot:header</h2>
      </template>
      <template #header>
        <h2>简写呀：#header</h2>
      </template>
    </navigation-link>

  </div>
</body>
<script src="https://cdn.bootcdn.net/ajax/libs/vue/2.6.9/vue.js"></script>
<script src="https://unpkg.com/element-ui/lib/index.js"></script>
<script>
  /**
   * 子组件
  */

  Vue.component('navigation-link', {
    data() {
      return {
      }
    },
    template: `
    <div>
      <h2>具名插槽</h2>
      <ul>
        <li>
          <p>【 将指定的内容放到指定的位置 】</p>
          <slot name="header">
            <el-link type="success">成功链接</el-link>
          </slot>
        </li>
      </ul>
    </div>

    `
  });

  var app = new Vue({
    el: '#app',
  });


</script>

</html>
```



### 作用域插槽

```html
<!DOCTYPE html>
<html lang="en">

<head>
  <meta charset="UTF-8">
  <meta http-equiv="X-UA-Compatible" content="IE=edge">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Document</title>
  <link rel="stylesheet" href="https://unpkg.com/element-ui/lib/theme-chalk/index.css">
  <style>

  </style>
</head>

<body>
  <div id="app">

    <!------ 作用域插槽 ------>
    <navigation-link>
      <!-- v-slot:插槽名="数据名（随便起）" -->
      <template  v-slot:header="slotProps">
        <p>{{slotProps.name}}</p>
      </template>
    </navigation-link>


  </div>
</body>
<script src="https://cdn.bootcdn.net/ajax/libs/vue/2.6.9/vue.js"></script>
<script src="https://unpkg.com/element-ui/lib/index.js"></script>
<script>
  /**
   * 子组件
  */
  Vue.component('navigation-link', {
    data() {
      return {
        user: {
          name: '李四',
          age: 18
        },
      }
    },
    template: `
    <div>
      <h2>作用域插槽</h2>
      <ul>
        <li>
          <p>【 让插槽内容能够访问子组件中才有的数据】</p>
          <slot v-bind:user="user">{{user.name}}</slot>
        </li>
      </ul>
    </div>

    `
  })

  var app = new Vue({
    el: '#app',
  });

</script>

</html>
```
